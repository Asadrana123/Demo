  <div class="box" id="box1">Box 1</div>
  <div class="box" id="box2">Box 2</div>
  <div class="box" id="box3">Box 3</div>
  <div class="box" id="box4">Box 4</div>
  <div class="box" id="box5">Box 5</div>

  <script>
    // 1. Select all elements you want to observe
    const boxes = document.querySelectorAll(".box");

    // 2. Create the IntersectionObserver
    const observer = new IntersectionObserver((entries, observer) => {
      entries.forEach(entry => {
        if(entry.isIntersecting){
          console.log(`${entry.target.id} is visible`);
          entry.target.style.background = "lightgreen"; // example: change color
          // Optional: stop observing after it becomes visible
          // observer.unobserve(entry.target);
        } else {
          // Reset background if element goes out of view
          entry.target.style.background = "lightblue";
        }
      });
    }, {
      root: null,       // viewport is root
      rootMargin: "0px",
      threshold: 0.5    // 50% of element must be visible
    });

    // 3. Observe all boxes
    boxes.forEach(box => observer.observe(box));
  </script>
